What is Material UI?

It's a React UI Component Library

Key Benefits of Material UI

  • Time-saving
  • Highly customizable
  • Consistent design

Basic Example


			  import { Button } from '@material-ui/core';
			  
			  <Button color="secondary">Secondary</Button>

Typography Example


import React from 'react';
import Typography from '@mui/material/Typography';
			  
function MyComponent() {
  return (
    <div>
      <Typography variant="h1" component="h2">Welcome</Typography>
	    <Typography variant="body1">Some text</Typography>
    </div>
	);
}

Customizing Your Component


import { makeStyles } from '@material-ui/core';
				
const useStyles = makeStyles({
  title: {
    marginBottom: 20,
  }
});
				
export default function Custom() {
  const classes = useStyles();
				
  return (
    <Typography
      className={classes.title}
	    variant="h6" 
	  >
	  Your Customized Title
	  </Typography>
  );
}

Customizing Your Theme


import {
  createMuiTheme,
  ThemeProvider } from '@material-ui/core';
			  
const theme = createMuiTheme({
  palette: {
    primary: {
      main: '#fefefe'
    },
  },
  typography: {
    fontFamily: 'Quicksand',
  }
});
			  
function App() {
  return (
    <ThemeProvider theme={theme}>
      <Router>
        
      </Router>
    </ThemeProvider>
  );
}
